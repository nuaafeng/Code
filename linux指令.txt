tmux ls                                查看当前所有的Tmux会话
tmux a -t 0        链接到某个会话       使用会话编号
tmux a -t <session-name>               使用会话名称
tmux kill-session -t 0                 使用编号
tmux kill-session -t <session-name>     使用会话名称

tmux switch -t <session-name>          使用会话名称
